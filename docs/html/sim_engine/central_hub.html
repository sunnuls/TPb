Central Hub for Multi-Agent State Synchronization.

This module provides WebSocket-based coordination for multi-agent simulations, enabling secure state synchronization and collective decision-making for game theory research.

Educational Use Only: Designed for controlled research environments to study multi-agent coordination patterns. Not intended for production gaming or real-money applications.

 80class CentralHub:
 81    """
 82    Central coordination hub for multi-agent simulations.
 83    
 84    Features (Пункт 1 & Подпункты 1.1, 1.2):
 85    - WebSocket-based agent connectivity
 86    - Encrypted state data transmission for secure research
 87    - Shared state synchronization for 2+ agents
 88    - Collective probability recalculation
 89    - Conflict avoidance in multi-agent scenarios
 90    - Reconnect and heartbeat mechanisms
 91    - Support for multiple concurrent simulation environments
 92    
 93    Educational Note:
 94        This hub enables research into coordination problems in game theory,
 95        studying how agents can cooperate without revealing full private
 96        information while avoiding conflicting actions.
 97    """

 99    def __init__(
100        self,
101        host: str = "localhost",
102        port: int = 8765,
103        encryption_key: Optional[bytes] = None,
104        heartbeat_interval: float = 30.0,
105        heartbeat_timeout: float = 90.0
106    ):
107        """
108        Initialize central hub.
109        
110        Args:
111            host: Host to bind WebSocket server
112            port: Port for WebSocket server
113            encryption_key: Fernet encryption key (generated if None)
114            heartbeat_interval: Seconds between heartbeat checks
115            heartbeat_timeout: Seconds before agent considered disconnected
116        """

118        self.port = port
119        
120        # Encryption for secure research data flow
121        self.encryption_key = encryption_key or Fernet.generate_key()
122        self.cipher = Fernet(self.encryption_key)
123        
124        # Connected agents
125        self.agents: Dict[str, AgentConnection] = {}
126        self._connections = self.agents  # Alias for backward compatibility
127        
128        # Environment groupings (agents in same simulation environment)
129        self.environments: Dict[str, Set[str]] = {}
130        
131        # Roadmap2 Phase 2: Sessions for card sharing
132        self._sessions: Dict[str, Dict[str, Any]] = {}
133        
134        # Heartbeat configuration
135        self.heartbeat_interval = heartbeat_interval
136        self.heartbeat_timeout = heartbeat_timeout
137        
138        # Message queue for processing
139        self.message_queue: asyncio.Queue = asyncio.Queue()

141        # Server instance
142        self.server: Optional[websockets.WebSocketServer] = None

144    def encrypt_state(self, state_data: Dict[str, Any]) -> str:
145        """
146        Encrypt state data for secure transmission.
147        
148        Educational Note:
149            Encryption ensures private agent information remains confidential
150            during research, modeling real-world privacy constraints.
151        """
152        json_data = json.dumps(state_data)
153        encrypted = self.cipher.encrypt(json_data.encode())
154        return encrypted.decode()

156    def decrypt_state(self, encrypted_data: str) -> Dict[str, Any]:
157        """Decrypt received state data."""
158        decrypted = self.cipher.decrypt(encrypted_data.encode())
159        return json.loads(decrypted.decode())

162        self,

167        """

201    async def handle_state_sync(
202        self,
203        agent_id: str,
204        message: AgentStateMessage
205    ) -> None:
206        """
207        Handle state synchronization request (Подпункт 1.1).
208        
209        Logic:
210        - If 2+ agents in environment, broadcast state update
211        - Recalculate collective probabilities
212        - Check for conflicting actions
213        
214        Educational Note:
215            This implements coordination logic to prevent agents from
216            making mutually incompatible decisions in shared simulations.
217        """
218        if agent_id not in self.agents:
219            return
220        
221        agent = self.agents[agent_id]
222        environment_id = message.environment_id or agent.environment_id
223        
224        if not environment_id:
225            return
226        
227        # Update agent's state
228        agent.state_data = message.state_data
229        
230        # Get agents in same environment
231        agents_in_env = self.environments.get(environment_id, set())
232        
233        # If 2+ agents, perform collective synchronization
234        if len(agents_in_env) >= 2:
235            await self._sync_environment_state(environment_id, agents_in_env)

237    async def _sync_environment_state(
238        self,
239        environment_id: str,
240        agent_ids: Set[str]
241    ) -> None:
242        """
243        Synchronize state across multiple agents in same environment.
244        
245        Performs:
246        - Collective probability aggregation
247        - Conflict detection
248        - State broadcast to all agents in environment
249        """
250        # Collect all agent states
251        collective_state = {}
252        planned_actions = []
253        
254        for agent_id in agent_ids:
255            if agent_id not in self.agents:
256                continue
257            
258            agent = self.agents[agent_id]
259            collective_state[agent_id] = agent.state_data
260            
261            # Collect planned actions for conflict detection
262            if "planned_action" in agent.state_data:
263                planned_actions.append({
264                    "agent_id": agent_id,
265                    "action": agent.state_data["planned_action"]
266                })
267        
268        # Detect conflicts (Подпункт 1.1: Avoid conflicting actions)
269        conflicts = self._detect_action_conflicts(planned_actions)
270        
271        # Recalculate collective probabilities
272        collective_probs = self._calculate_collective_probabilities(collective_state)
273        
274        # Broadcast synchronized state to all agents
275        sync_message = {
276            "message_type": MessageType.STATE_SYNC.value,
277            "environment_id": environment_id,
278            "timestamp": time.time(),
279            "collective_probabilities": collective_probs,
280            "conflicts_detected": conflicts,
281            "agent_count": len(agent_ids)
282        }
283        
284        # Send to all agents in environment
285        for agent_id in agent_ids:
286            if agent_id in self.agents:
287                agent = self.agents[agent_id]
288                try:
289                    await agent.websocket.send(json.dumps(sync_message))
290                except websockets.exceptions.ConnectionClosed:
291                    # Handle disconnection
292                    await self._handle_disconnect(agent_id)

294    def _detect_action_conflicts(
295        self,
296        planned_actions: List[Dict[str, Any]]
297    ) -> List[str]:
298        """
299        Detect conflicting actions between agents.
300        
301        Educational Note:
302            In multi-agent simulations, certain action combinations may be
303            logically incompatible (e.g., multiple agents claiming same resource).
304            This detects such conflicts for research purposes.
305        """
306        conflicts = []
307        
308        # Example conflict detection: multiple agents planning same exclusive action
309        exclusive_actions = {}
310        
311        for action_data in planned_actions:
312            action = action_data["action"]
313            action_type = action.get("type")
314            
315            # Check for exclusive resource claims
316            if action_type == "increment" and action.get("exclusive_resource"):
317                resource = action["exclusive_resource"]
318                
319                if resource in exclusive_actions:
320                    conflicts.append(
321                        f"Conflict: {action_data['agent_id']} and "
322                        f"{exclusive_actions[resource]} both claim {resource}"
323                    )
324                else:
325                    exclusive_actions[resource] = action_data['agent_id']
326        
327        return conflicts

329    def _calculate_collective_probabilities(
330        self,
331        collective_state: Dict[str, Dict[str, Any]]
332    ) -> Dict[str, float]:
333        """
334        Recalculate probabilities based on collective agent information.
335        
336        Educational Note:
337            Aggregates information from multiple agents to produce more
338            accurate probability estimates for research scenarios.
339        """
340        # Simple aggregation: average equity estimates
341        equities = []
342        confidences = []
343        
344        for agent_id, state in collective_state.items():
345            if "equity" in state:
346                equities.append(state["equity"])
347            if "confidence" in state:
348                confidences.append(state["confidence"])
349        
350        result = {}
351        
352        if equities:
353            result["average_equity"] = sum(equities) / len(equities)
354            result["equity_variance"] = sum((e - result["average_equity"]) ** 2 for e in equities) / len(equities)
355        
356        if confidences:
357            result["average_confidence"] = sum(confidences) / len(confidences)
358        
359        return result

361    async def handle_heartbeat(self, agent_id: str) -> None:
362        """
363        Handle heartbeat from agent (Подпункт 1.2).
364        
365        Educational Note:
366            Heartbeats ensure agents remain connected during long simulations,
367            enabling automatic recovery from network issues.
368        """
369        if agent_id in self.agents:
370            self.agents[agent_id].last_heartbeat = time.time()
371            
372            # Send heartbeat acknowledgment
373            response = {
374                "message_type": MessageType.HEARTBEAT.value,
375                "status": "ack",
376                "timestamp": time.time()
377            }
378            
379            try:
380                await self.agents[agent_id].websocket.send(json.dumps(response))
381            except websockets.exceptions.ConnectionClosed:
382                await self._handle_disconnect(agent_id)

384    async def _handle_disconnect(self, agent_id: str) -> None:
385        """Handle agent disconnection."""
386        if agent_id not in self.agents:
387            return
388        
389        agent = self.agents[agent_id]
390        
391        # Remove from environment
392        if agent.environment_id and agent.environment_id in self.environments:
393            self.environments[agent.environment_id].discard(agent_id)
394            
395            # Notify other agents in environment
396            if self.environments[agent.environment_id]:
397                await self._broadcast_agent_status(
398                    agent.environment_id,
399                    agent_id,
400                    AgentStatus.DISCONNECTED
401                )
402        
403        # Remove agent
404        del self.agents[agent_id]

427    async def _heartbeat_monitor(self) -> None:
428        """
429        Monitor agent heartbeats and disconnect stale agents (Подпункт 1.2).
430        
431        Educational Note:
432            Automatic cleanup ensures simulations don't get stuck waiting
433            for disconnected agents.
434        """
435        while True:
436            await asyncio.sleep(self.heartbeat_interval)
437            
438            current_time = time.time()
439            disconnected_agents = []
440            
441            for agent_id, agent in self.agents.items():
442                if current_time - agent.last_heartbeat > self.heartbeat_timeout:
443                    disconnected_agents.append(agent_id)
444            
445            # Clean up disconnected agents
446            for agent_id in disconnected_agents:
447                await self._handle_disconnect(agent_id)

449    async def handle_client(
450        self,
451        websocket: websockets.WebSocketServerProtocol
452    ) -> None:
453        """
454        Handle WebSocket client connection.
455        
456        Main message loop for agent communication.
457        """
458        agent_id: Optional[str] = None
459        
460        try:
461            async for message_raw in websocket:
462                try:
463                    message_data = json.loads(message_raw)
464                    message = AgentStateMessage(**message_data)
465                    
466                    # Handle registration
467                    if message.message_type == MessageType.REGISTER:
468                        agent_id = message.agent_id
469                        await self.register_agent(
470                            websocket,
471                            agent_id,
472                            message.environment_id
473                        )
474                    
475                    # Subsequent messages require registration
476                    elif agent_id is None:
477                        error_msg = {
478                            "message_type": MessageType.ERROR.value,
479                            "error": "Agent not registered"
480                        }
481                        await websocket.send(json.dumps(error_msg))
482                        continue
483                    
484                    # Handle state sync
485                    elif message.message_type == MessageType.STATE_SYNC:
486                        await self.handle_state_sync(agent_id, message)
487                    
488                    # Handle heartbeat
489                    elif message.message_type == MessageType.HEARTBEAT:
490                        await self.handle_heartbeat(agent_id)
491                    
492                except Exception as e:
493                    error_msg = {
494                        "message_type": MessageType.ERROR.value,
495                        "error": str(e)
496                    }
497                    await websocket.send(json.dumps(error_msg))
498        
499        except websockets.exceptions.ConnectionClosed:
500            pass
501        
502        finally:
503            if agent_id:
504                await self._handle_disconnect(agent_id)

526    async def share_cards(
527        self,
528        environment_id: str,
529        agent_id: str,
530        hole_cards: List[str]
531    ) -> Optional[Dict[str, Any]]:
532        """
533        Share agent's hole cards with the collective (Roadmap2 Phase 2).
534        
535        When ≥3 agents in a session, they pool their hole cards to calculate
536        collective equity vs dummy opponent and remaining deck.
537        
538        Args:
539            environment_id: Session/table identifier
540            agent_id: Agent sharing cards
541            hole_cards: List of 2 cards (e.g., ["As", "Kh"])
542            
543        Returns:
544            Dict with collective_known_cards, collective_equity, dummy_range
545            
546        Educational Note:
547            This simulates HIVE strategy where multiple agents coordinate
548            by sharing private information to gain collective advantage.
549            For academic study of coordination mechanisms only.
550        """
551        # Get or create session
552        if environment_id not in self._sessions:
553            self._sessions[environment_id] = {}
554        session = self._sessions[environment_id]
555            
556        # Store cards in session
557        if "hole_cards" not in session:
558            session["hole_cards"] = {}
559        session["hole_cards"][agent_id] = hole_cards
560        
561        # Calculate collective knowledge when ≥3 agents
562        agents_in_session = [
563            conn for conn in self._connections.values()
564            if conn.environment_id == environment_id
565        ]
566        
567        if len(agents_in_session) < 3:
568            return None
569            
570        # Collect all known cards
571        collective_cards = []
572        for aid, cards in session["hole_cards"].items():
573            collective_cards.extend(cards)
574            
575        # Calculate equity (simplified for demonstration)
576        equity = self._calculate_collective_equity(
577            collective_cards,
578            environment_id
579        )
580        
581        return {
582            "collective_known_cards": collective_cards,
583            "collective_equity": equity,
584            "dummy_range": session.get("dummy_range", "random"),
585            "agent_count": len(agents_in_session)
586        }

588    def _calculate_collective_equity(
589        self,
590        known_cards: List[str],
591        environment_id: str
592    ) -> float:
593        """
594        Calculate collective equity for HIVE group.
595        
596        Educational simplification:
597        - More known cards = higher equity estimate
598        - Real implementation would use poker equity calculator
599        - For research: studies how information pooling affects decisions
600        
601        Args:
602            known_cards: All cards known to the collective
603            environment_id: Session identifier
604            
605        Returns:
606            Equity percentage (0.0 to 1.0)
607        """
608        session = self._sessions.get(environment_id, {})
609        board = session.get("board", [])
610        
611        # Simplified equity calculation
612        # More known cards + coordinated agents = higher equity
613        base_equity = 0.5
614        card_bonus = len(known_cards) * 0.03  # +3% per known card
615        board_bonus = len(board) * 0.02  # +2% per board card
616        
617        equity = min(0.95, base_equity + card_bonus + board_bonus)
618        
619        return equity

341        # Simple aggregation: average equity estimates

343        confidences = []

357        if confidences:
358            result["average_confidence"] = sum(confidences) / len(confidences)
359
